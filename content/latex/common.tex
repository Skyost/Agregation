\newcommand{\@textbox}[1]{
	\tikz[baseline=(box.base)]{
		\node[
		fill=black,
		rounded corners=\@boxradius,
		inner xsep=0.5em,
		inner ysep=0.25em,
		outer sep=0pt,
		text=white
		] (box) {\MakeUppercase{\sffamily\bfseries #1}};
	}%
}

\NewDocumentCommand{\@headertextbox}{m}{%
	\tikz[baseline=(box.base)]{%
		\node[
		inner xsep=0.5em,
		inner ysep=0.25em,
		outer sep=0pt,
		font=\sffamily\bfseries\scriptsize
		] (box) {\MakeUppercase{#1}};
		\fill[black]
		(box.south west) --
		(box.south east) --
		([yshift=-\@boxradius]box.north east)
		.. controls (box.north east) .. ([xshift=-\@boxradius]box.north east) --
		([xshift=\@boxradius]box.north west)
		.. controls (box.north west) .. ([yshift=-\@boxradius]box.north west) --
		cycle;
		\node[
		inner xsep=0.5em,
		inner ysep=0.25em,
		outer sep=0pt,
		text=white,
		font=\sffamily\bfseries\scriptsize
		] at (box.center) {\MakeUppercase{#1}};
	}%
}

\NewDocumentCommand{\@theoremboxtitle}{m m}{%
	\@textbox{%
		\scriptsize#1%

	}
}

	\begin{tikzpicture}%
		\draw[fill=black] (0,0) circle (1.5pt);
		\node at (0.15,0) [right] {\sffamily\scriptsize\bfseries\upshape\MakeUppercase{#1} #2};
	\end{tikzpicture}%
}

	}
